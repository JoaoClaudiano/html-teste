// generate-errors.js CORRIGIDO const fs = require("fs"); // Ler dados dos erros (nome corrigido) const errors = JSON.parse(fs.readFileSync("erros.json", "utf8")); // Template HTML básico const template = ` {{TITLE}} — Testador de HTML ← Voltar para o Testador

{{TITLE}}

Problema: {{PROBLEM}}

Impacto:

    {{IMPACT_LIST}}

Solução:

{{SOLUTION}}
`; if (!fs.existsSync("erros")) { fs.mkdirSync("erros"); } errors.forEach(err => { const impactList = err.impact .map(item => `
  • ${item}
    • `) .join(''); let page = template .replace(/{{TITLE}}/g, err.title) .replace(/{{DESCRIPTION}}/g, err.description) .replace(/{{PROBLEM}}/g, err.problem) .replace(/{{IMPACT_LIST}}/g, impactList) .replace(/{{SOLUTION}}/g, err.solution); fs.writeFileSync(`erros/${err.slug}.html`, page); }); console.log(`Páginas de erro geradas: ${errors.length}`);
